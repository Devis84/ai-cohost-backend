  <title>Dashboard</title>

</head>

<body>


<h2>📊 Conversations</h2>


<button onclick="load()">Refresh</button>


<div id="data"></div>


<script>

async function load() {

  const res = await fetch("/conversations");

  const data = await res.json();


  document.getElementById("data").innerHTML =

    data.map(m => `

      <p>

        <b>${m.phone}</b> (${m.role})<br/>

        ${m.message}

      </p>

      <hr/>

    `).join("");

}


load();

</script>


</body>
