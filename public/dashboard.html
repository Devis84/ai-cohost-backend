<h2>📊 Conversations</h2>


<button onclick="load()">🔄 Refresh</button>


<div id="data"></div>


<h3>👤 Host Control</h3>


<input id="phone" placeholder="Phone">

<input id="message" placeholder="Message">


<button onclick="sendHost()">Send</button>

<button onclick="takeover(true)">Takeover ON</button>

<button onclick="takeover(false)">Takeover OFF</button>


<script>

async function load() {

  const res = await fetch("/conversations");

  const data = await res.json();


  document.getElementById("data").innerHTML =

    data.map(m => `

      <div class="msg ${m.role}">

        <b>${m.phone}</b> (${m.role})<br/>

        ${m.message}

      </div>

    `).join("");

}


async function sendHost() {

  await fetch("/host-send", {

    method: "POST",

    headers: { "Content-Type": "application/json" },

    body: JSON.stringify({

      phone: document.getElementById("phone").value,

      message: document.getElementById("message").value

    })

  });

}


async function takeover(active) {

  await fetch("/takeover", {

    method: "POST",

    headers: { "Content-Type": "application/json" },

    body: JSON.stringify({

      phone: document.getElementById("phone").value,

      active

    })

  });

}


// auto refresh

setInterval(load, 3000);


load();

</script>
