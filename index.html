const API = "https://ai-cohost-backend.onrender.com";


// ===== LOAD TASKS =====

async function loadTasks() {

  const res = await fetch(

    `${SUPABASE_URL}/rest/v1/cleaning_tasks?select=*`,

    {

      headers: {

        apikey: SUPABASE_KEY,

        Authorization: `Bearer ${SUPABASE_KEY}`

      }

    }

  );


  const data = await res.json();

  const container = document.getElementById("tasks");

  container.innerHTML = "";


  data.forEach(task => {

    const div = document.createElement("div");


    div.innerHTML = `

      <hr>

      <b>${task.cleaning_date}</b><br>

      ID: ${task.id}<br>


      <input placeholder="Cleaner" id="c-${task.id}">

      <input placeholder="Start" id="s-${task.id}">

      <input placeholder="End" id="e-${task.id}">

      <input placeholder="Rate" id="r-${task.id}">

      <input placeholder="Notes" id="n-${task.id}">


      <button onclick="update('${task.id}')">UPDATE</button>

    `;


    container.appendChild(div);

  });

}


// ===== UPDATE =====

async function update(id) {

  const cleaner = document.getElementById(`c-${id}`).value;

  const start = document.getElementById(`s-${id}`).value;

  const end = document.getElementById(`e-${id}`).value;

  const rate = document.getElementById(`r-${id}`).value;

  const notes = document.getElementById(`n-${id}`).value;


  await fetch(`${API}/cleaning/update`, {

    method: "POST",

    headers: {"Content-Type": "application/json"},

    body: JSON.stringify({

      id,

      cleaner,

      start_time: start,

      end_time: end,

      hourly_rate: Number(rate),

      notes,

      status: "done"

    })

  });


  alert("Updated");

  loadTasks();

}


// INIT
